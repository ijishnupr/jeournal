Welcome to Trade Journal

No OAuth or API keys needed — just paste a script into your Google Sheet and you're ready.

  1. Open your Google Sheet → Extensions → Apps Script
    Delete any existing code in the editor
  2. Paste this script and click Save (Ctrl+S) 
    const SHEET_NAME = 'Trades';
const HEADERS = ['Timestamp','Date','Entry Time','Exit Time','Time Taken','Symbol','Market','Direction','Entry Price','Exit Price','Quantity','Gross P&L','P&L %','Net P&L','Stop Loss','Take Profit','Charges','Strategy','Notes','Mistakes','Rating','Image URL','Drive File ID'];

function doGet(e) {
  try {
    const s = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
    if (!s || s.getLastRow() <= 1) return out({ trades: [] });
    const rows = s.getDataRange().getValues();
    rows.shift();
    const clean = rows.map(row => row.map(cell => {
      if (cell instanceof Date) {
        if (cell.getFullYear() <= 1900) {
          return String(cell.getHours()).padStart(2,'0') + ':' + String(cell.getMinutes()).padStart(2,'0');
        }
        return cell.getFullYear() + '-' + String(cell.getMonth()+1).padStart(2,'0') + '-' + String(cell.getDate()).padStart(2,'0');
      }
      return cell;
    }));
    return out({ trades: clean });
  } catch(e) { return out({ error: e.message }); }
}

function doPost(e) {
  try {
    const d = JSON.parse(e.postData.contents);
    if (d.type === 'image') {
      const f = DriveApp.createFile(Utilities.newBlob(Utilities.base64Decode(d.base64), d.mimeType, d.filename));
      f.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
      return out({ success:true, fileId:f.getId(), url:'https://drive.google.com/file/d/'+f.getId()+'/view' });
    }
    if (d.type === 'update') {
      const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
      if (!sheet) return out({ error: 'Sheet not found' });
      const values = sheet.getDataRange().getValues();
      for (let i = 1; i < values.length; i++) {
        if (String(values[i][0]) === String(d.timestamp)) {
          sheet.getRange(i+1, 1, 1, d.row.length).setValues([d.row]);
          return out({ success: true });
        }
      }
      return out({ error: 'Trade not found' });
    }
    if (d.type === 'trade') {
      const ss = SpreadsheetApp.getActiveSpreadsheet();
      let sheet = ss.getSheetByName(SHEET_NAME);
      if (!sheet) sheet = ss.insertSheet(SHEET_NAME);
      if (sheet.getLastRow() === 0) sheet.appendRow(HEADERS);
      sheet.appendRow(d.row);
      return out({ success: true });
    }
    return out({ error: 'Unknown request type' });
  } catch(e) { return out({ error: e.message }); }
}

function out(o) { return ContentService.createTextOutput(JSON.stringify(o)).setMimeType(ContentService.MimeType.JSON); }
  3. Deploy → Click Deploy button (top right) → New deployment
    Set type to Web app → Execute as: Me → Who has access: Anyone → Deploy
    Authorize when prompted — this lets the script write to your own sheet
